class SemanticManifold:
    """Semantic Emotional Manifold v2.0 with S-curve adaptive inertia."""

    def __init__(self, model_name="microsoft/harrier-oss-v1-0.6b",
                 alpha_base=ALPHA_BASE, top_k=TOP_K,
                 default_weight=DEFAULT_WEIGHT,
                 momentum_window=MOMENTUM_WINDOW):
        self.alpha_base = alpha_base
        self.top_k = top_k
        self.default_weight = default_weight
        self.model_name = model_name
        self.momentum_window = momentum_window

        from sentence_transformers import SentenceTransformer
        self._model = SentenceTransformer(model_name)

        self._anchor_vectors = {}
        self._anchor_texts = {}
        for name, data in ANCHOR_TEXTS.items():
            self._anchor_texts[name] = data["text"]
            self._anchor_vectors[name] = self._model.encode(
                data["text"], normalize_embeddings=True)

        self.weights = {name: default_weight for name in ANCHOR_TEXTS}
        self._momentum = {name: deque(maxlen=momentum_window) for name in ANCHOR_TEXTS}
        self._volatility_history = deque(maxlen=VOLATILITY_WINDOW)
        self._previous_weights = None
        self.volatility = 0.0
        self.stress = 0.0

    def _compute_alpha(self, name, resonance):
        """S-curve adaptive alpha: resistance -> flow -> saturation."""
        current = self.weights[name]
        delta_direction = resonance - current
        history = self._momentum[name]

        # Momentum
        if len(history) == 0:
            momentum = 0
        else:
            same_direction = sum(
                1 for d in history
                if (d > 0 and delta_direction > 0) or (d < 0 and delta_direction < 0))
            momentum = same_direction / len(history)

        # Direction: resistance vs flow
        if len(history) >= 2:
            recent_trend = sum(history) / len(history)
            opposing = ((recent_trend > 0.001 and delta_direction < -0.001) or
                       (recent_trend < -0.001 and delta_direction > 0.001))
        else:
            opposing = False

        if opposing:
            direction_factor = RESISTANCE_FACTOR
        else:
            direction_factor = FLOW_BASE + (momentum * FLOW_BOOST)

        # Saturation near extremes
        distance_to_extreme = min(current, 1.0 - current)
        saturation_factor = distance_to_extreme ** (1.0 / SATURATION_STRENGTH)
        saturation_factor = max(0.1, saturation_factor * 2)

        alpha_effective = self.alpha_base * direction_factor * saturation_factor
        return max(0.01, min(0.5, alpha_effective))

    def _cosine_similarity(self, v1, v2):
        return float(np.dot(v1, v2))

    @staticmethod
    def _clamp(value, min_val=0.0, max_val=1.0):
        return max(min_val, min(max_val, value))

    def update(self, message):
        """Update manifold with S-curve adaptive alpha."""
        self._previous_weights = dict(self.weights)
        v_input = self._model.encode(message, normalize_embeddings=True)
        total_abs_delta = 0.0

        for name, v_anchor in self._anchor_vectors.items():
            resonance = self._cosine_similarity(v_input, v_anchor)
            alpha = self._compute_alpha(name, resonance)
            current = self.weights[name]
            delta = (resonance - current) * alpha
            self.weights[name] = self._clamp(current + delta)
            self._momentum[name].append(delta)
            total_abs_delta += abs(delta)

        avg_delta = total_abs_delta / len(ANCHOR_TEXTS)
        self._volatility_history.append(avg_delta)
        self.volatility = sum(self._volatility_history) / len(self._volatility_history)
        self.stress = max(0.0, (self.volatility - STRESS_THRESHOLD) / STRESS_THRESHOLD)
        return dict(self.weights)

    def get_equation(self):
        sorted_dims = sorted(self.weights.items(), key=lambda x: x[1], reverse=True)
        terms = [f"({v:.2f} x V_{n})" for n, v in sorted_dims[:self.top_k]]
        return "E_state = " + " + ".join(terms)

    def to_prompt_context(self):
        equation = self.get_equation()
        sorted_dims = sorted(self.weights.items(), key=lambda x: x[1], reverse=True)
        active = ", ".join(f"{n}({v:.2f})" for n, v in sorted_dims[:self.top_k])
        result = f"{equation}\nActive poles: {active}"
        if self.stress > 0.5:
            result += f"\nEmotional volatility: elevated ({self.stress:.2f})"
        return result

    def get_full_state(self):
        sorted_dims = sorted(self.weights.items(), key=lambda x: x[1], reverse=True)
        lines = []
        for name, value in sorted_dims:
            history = self._momentum[name]
            if len(history) >= 2:
                trend = sum(history) / len(history)
                arrow = " ^" if trend > 0.005 else " v" if trend < -0.005 else " >"
            else:
                arrow = ""
            lines.append(f"{name}: {value:.4f}{arrow}")
        if self.stress > 0.1:
            lines.append(f"Volatility: {self.volatility:.4f} (stress: {self.stress:.2f})")
        return "\n".join(lines)

    def get_delta(self, prev_state=None, trigger=None):
        reference = prev_state or self._previous_weights
        if reference is None:
            raise ValueError("No previous state. Call update() first.")
        delta = {}
        for name in self.weights:
            current = self.weights[name]
            previous = reference.get(name, self.default_weight)
            diff = current - previous
            direction = "=" if abs(diff) < 0.001 else "+" if diff > 0 else "-"
            history = self._momentum[name]
            mom = 0.0
            if len(history) >= 2:
                same = sum(1 for d in history if (d > 0) == (diff > 0))
                mom = same / len(history)
            entry = {"direction": direction, "magnitude": round(abs(diff), 4),
                     "previous": round(previous, 4), "current": round(current, 4),
                     "momentum": round(mom, 2)}
            if trigger:
                entry["trigger"] = trigger
            delta[name] = entry
        return delta

    def get_delta_summary(self, prev_state=None, trigger=None, threshold=0.003):
        delta = self.get_delta(prev_state=prev_state, trigger=trigger)
        sig = {k: v for k, v in delta.items() if v["magnitude"] >= threshold}
        if not sig:
            return "No significant emotional shift."
        sorted_c = sorted(sig.items(), key=lambda x: x[1]["magnitude"], reverse=True)
        lines = []
        for name, d in sorted_c:
            arr = "^" if d["direction"] == "+" else "v" if d["direction"] == "-" else ">"
            mom = f" [momentum:{d['momentum']:.0%}]" if d["momentum"] > 0.5 else ""
            lines.append(f"  {arr} {name}: {d['previous']:.3f} -> {d['current']:.3f} "
                        f"(D{d['direction']}{d['magnitude']:.4f}){mom}")
        header = "Emotional trajectory:"
        if trigger:
            header += f" [trigger: {trigger[:60]}]"
        if self.stress > 0.3:
            header += f" ! stress: {self.stress:.2f}"
        return header + "\n" + "\n".join(lines)

    def get_stress_report(self):
        if self.stress < 0.1:
            status = "Stable"
        elif self.stress < 0.5:
            status = "Mild turbulence"
        elif self.stress < 1.0:
            status = "Elevated stress"
        else:
            status = "High stress"
        report = f"Volatility: {self.volatility:.4f}\n"
        report += f"Stress: {self.stress:.2f}\n"
        report += f"Status: {status}\n"
        dim_vol = {}
        for name, history in self._momentum.items():
            if len(history) >= 2:
                dim_vol[name] = sum(abs(d) for d in history) / len(history)
        if dim_vol:
            top = sorted(dim_vol.items(), key=lambda x: x[1], reverse=True)[:3]
            report += "Most active: " + ", ".join(f"{n}(D{v:.4f})" for n, v in top)
        return report

    def save(self, filepath):
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        mom_data = {name: list(h) for name, h in self._momentum.items()}
        data = {"timestamp": timestamp, "version": "2.0",
                "alpha_base": self.alpha_base, "weights": self.weights,
                "momentum": mom_data, "volatility": self.volatility,
                "stress": self.stress, "equation": self.get_equation()}
        with open(filepath, "w", encoding="utf-8") as f:
            json.dump(data, f, indent=2, ensure_ascii=False)
        txt_path = os.path.splitext(filepath)[0] + ".txt"
        with open(txt_path, "a", encoding="utf-8") as f:
            f.write(f"\n[{timestamp}]\n")
            for name, value in sorted(self.weights.items(), key=lambda x: x[1], reverse=True):
                history = self._momentum[name]
                trend = sum(history) / len(history) if history else 0
                arr = "^" if trend > 0.005 else "v" if trend < -0.005 else ">"
                f.write(f"{name}: {value:.4f} {arr}\n")
            f.write(f"# {self.get_equation()}\n")
            if self.stress > 0.1:
                f.write(f"# Stress: {self.stress:.2f}\n")

    def load(self, filepath):
        with open(filepath, "r", encoding="utf-8") as f:
            data = json.load(f)
        loaded = data.get("weights", {})
        for name in ANCHOR_TEXTS:
            if name not in loaded:
                loaded[name] = self.default_weight
        self.weights = loaded
        mom_data = data.get("momentum", {})
        for name in ANCHOR_TEXTS:
            if name in mom_data:
                self._momentum[name] = deque(mom_data[name], maxlen=self.momentum_window)
            else:
                self._momentum[name] = deque(maxlen=self.momentum_window)
        self.volatility = data.get("volatility", 0.0)
        self.stress = data.get("stress", 0.0)
        return dict(self.weights)

    def get_anchor_names(self):
        return list(ANCHOR_TEXTS.keys())

    def get_anchor_text(self, name):
        if name not in self._anchor_texts:
            raise KeyError(f"Unknown anchor: {name}")
        return self._anchor_texts[name]

    def get_momentum_summary(self):
        summary = {}
        for name, history in self._momentum.items():
            if len(history) < 2:
                summary[name] = "neutral"
            else:
                trend = sum(history) / len(history)
                summary[name] = "rising" if trend > 0.005 else "falling" if trend < -0.005 else "stable"
        return summary

    def reset(self):
        self.weights = {name: self.default_weight for name in ANCHOR_TEXTS}
        self._momentum = {name: deque(maxlen=self.momentum_window) for name in ANCHOR_TEXTS}
        self._volatility_history.clear()
        self._previous_weights = None
        self.volatility = 0.0
        self.stress = 0.0

    def __repr__(self):
        eq = self.get_equation()
        s = f", stress={self.stress:.2f}" if self.stress > 0.1 else ""
        return f"SemanticManifold(v2, a={self.alpha_base}{s}, {eq})"
